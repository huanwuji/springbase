253 
254 		if(xeno.common.lang.Data.isNull(str) && xeno.common.lang.Data.isNull(search)) {
255 			return replace;
256 
257 		} else if(xeno.common.lang.Data.isNull(str) || xeno.common.lang.Data.isNull(search)) {
258 			return str;
259 		}
260 
261 		// Could not use the regex to replace all search String, because the "search" String may contains some regex reserved characters.
262 		// For an example: xeno.common.util.StringUtils.replace("m{0}ya", "{0}", "a") will cause error, becuase "{" is a regex reserved character.
263 		// To ensure this function to work correct, you must input: xeno.common.util.StringUtils.replace("m{0}ya", "\\{0\\}", "a"), but this is not what we want.
264 		// return str.replace(new RegExp(search, "gm"), replace);
265 
266 		var found = false;
267 		var sLen = str.length;
268 		var rLen = search.length;
269 		var ret = "";
270 
271 		for(var i = 0; i < sLen; i += 1) {
272 
273 			if(str.charAt(i) === search.charAt(0)) {
274 				found = true;
275 
276 				for(var j = 0; j < rLen; j += 1) {
277 
278 					if(!(str.charAt(i + j) === search.charAt(j))) {
279 						found = false;
280 
281 						break;
282 					}
283 				}
284 
285 				if(found) {
286 					ret += "" + replace;
287 
288 					i = i + rLen - 1;
289 
290 					continue;
291 				}
292 			}
293 
294 			ret += str.charAt(i);
295 		}
296 
297 		return ret;
298 	};
