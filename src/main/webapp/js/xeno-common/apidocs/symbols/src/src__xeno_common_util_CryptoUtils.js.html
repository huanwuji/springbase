 18 		var bytes = [];
 19 
 20 		for(var i = 0; i < str.length; i += 1) {
 21 			bytes.push(str.charCodeAt(i) & 0xFF);
 22 		}

 27 	var _wordsToBytes = function(words) {
 28 		var bytes = [];
 29 
 30 		for(var i = 0; i < words.length * 32; i += 8) {
 31 			bytes.push((words[i >>> 5] >>> (24 - i % 32)) & 0xFF);
 32 		}
 33 
 34 		return bytes;
 35 	};

 37 	var _bytesToWords = function(bytes) {
 38 		var words = [];
 39 
 40 		for(var i = 0, b = 0; i < bytes.length; i += 1, b += 8) {
 41 			words[b >>> 5] |= (bytes[i] & 0xFF) << (24 - b % 32);
 42 		}
 43 
 44 		return words;
 45 	};

 47 	var _bytesToHex = function(bytes) {
 48 		var hex = [];
 49 
 50 		for(var i = 0; i < bytes.length; i += 1) {
 51 			hex.push((bytes[i] >>> 4).toString(16));
 52 			hex.push((bytes[i] & 0xF).toString(16));
 53 		}
 54 
 55 		return hex.join("");
 56 	};

 58 	var _rol = function(n, b) {
 59 		return (n << b) | (n >>> (32 - b));
 60 	};

 62 	var _endian = function(n) {
 63 
 64 		// If number given, swap endian.
 65 		if(xeno.common.lang.Data.isNumber(n)) {
 66 			return _rol(n, 8) & 0x00FF00FF | _rol(n, 24) & 0xFF00FF00;
 67 		}
 68 
 69 		// Else, assume array and swap all items.
 70 		for(var i = 0; i < n.length; i++) {
 71 			n[i] = _endian(n[i]);
 72 		}
 73 
 74 		return n;
 75 	};

202 	var _sha1 = function(str) {
203 		var message = _stringToBytes(unescape(encodeURIComponent(str)));
204 
205 		var m = _bytesToWords(message);
206 		var l = message.length * 8;
207 
208 		// Padding.
209 		m[l >> 5] |= 0x80 << (24 - l % 32);
210 		m[((l + 64 >>> 9) << 4) + 15] = l;
211 
212 		var w = [];
213 		var a = 1732584193, b = -271733879, c = -1732584194, d = 271733878, e = -1009589776;
214 		var aa = 0, bb = 0, cc = 0, dd = 0, ee = 0;

216 		for(var i = 0; i < m.length; i += 16) {
217 			aa = a;
