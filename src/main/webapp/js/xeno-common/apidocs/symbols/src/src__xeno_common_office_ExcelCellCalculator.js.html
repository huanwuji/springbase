1 /**
  2  * @class
  3  * <p>
  4  * This class has implemented some excel cell related calculations.
  5  * </p>
  6  * 
  7  * @description
  8  * <p>
  9  * The class constructor.
 10  * </p>
 11  * 
 12  * @since
 13  * 			1.0.0
 14  */
 15 xeno.common.office.ExcelCellCalculator = (function () {
 16 	var _address = new RegExp("^[A-Z]+[1-9]\\d*$", "i");
 17 	var _column = new RegExp("[A-Z]+", "i");
 18 
 19 	var _getChar = function(value) {
 20 		var mantissa = (value % 26) !== 0 ? value % 26 : 26;
 21 
 22 		if (value - mantissa >= 26) {
 23 			return _getChar(Math.floor((value - mantissa) / 26)) + String.fromCharCode(64 + mantissa);
 24 
 25 		} else {
 26 			return String.fromCharCode(64 + mantissa);
 27 		}
 28 	};
 29 
 30 	var _getInt = function(value) {
 31 		var val = value.toUpperCase();
 32 		var dec = 0;
 33 
 34 		for (var i = 0; i < val.length; i += 1) {
 35 			dec += (val.charCodeAt(val.length - 1 - i) - 64) * Math.pow(26, i);
 36 		}
 37 
 38 		return dec - 1;
 39 	};
 40 
 41 	// ------------------------------
 42 	// Private Function Definitions
 43 	// ------------------------------
 44 
 45 	var _getAddress = function(row, column, toUpperCase) {
 46 		var _row = Math.floor(row);
 47 		var _column = Math.floor(column);
 48 
 49 		var ret = _getChar(_column + 1) + (_row + 1);
 50 
 51 		return toUpperCase ? ret : ret.toLowerCase();
 52 	};
 53 
 54 	var _getIndex = function(address) {
 55 		var column = _column.exec(address)[0];
 56 		var row = xeno.common.util.StringUtils.remove(address, column);
 57 
 58 		return {
 59 			row: parseInt(row) - 1,
 60 			column: _getInt(column)
 61 		};
 62 	};
 63 
 64 	return {
 65 
 66 		/**
 67 		 * @description
 68 		 * <p>
 69 		 * Calculates and returns the excel cell address by the "row" and the "column" index.
 70 		 * </p>
 71 		 * <p>
 72 		 * The result will be returned in "A1" format, and the passes in arguments "row" and "column" are zero based.
 73 		 * </p>
 74 		 * <p>
 75 		 * The "row" or the "column" maybe a float value, the "Math.floor(n)" action will be processed first to convert it into an integer Number.
 76 		 * </p>
 77 		 * 
 78 		 * @example
 79 		 * <b>Examples</b>
 80 		 * 
 81 		 * xeno.common.office.ExcelCellCalculator.getAddress(8, 2) = "c9"
 82 		 * 
 83 		 * @param row
 84 		 * 			The row index.
 85 		 * @param column
 86 		 * 			The column index.
 87 		 * @param toUpperCase
 88 		 * 			A Boolean value to decide whether all characters in the return value are in upper case or not.<br/>
 89 		 * 			The default value is: false.
 90 		 * 
 91 		 * @return
 92 		 * 			The String.
 93 		 * 
 94 		 * @throws
 95 		 * 			The Error will be thrown when the "row" or the "column" is missing.<br/>
 96 		 * 			Or it will be thrown when the "row" or the "column" is not a Number.<br/>
 97 		 * 			Or it will be thrown when the "row" or the "column" is negative.
 98 		 * 
 99 		 * @since
100 		 * 			1.0.0
101 		 */
102 		getAddress: function(row, column, toUpperCase) {
103 			var _arguments = arguments.length;
104 
105 			xeno.common.lang.Data.ensures(_arguments > 0, "The 'row' is required, index: 0");
106 			xeno.common.lang.Data.ensures(_arguments > 1, "The 'column' is required, index: 1");
107 			xeno.common.lang.Data.ensures(xeno.common.lang.Data.isNumber(row), "The 'row' is not a Number");
108 			xeno.common.lang.Data.ensures(xeno.common.lang.Data.isNumber(column), "The 'column' is not a Number");
109 
110 			var _toUpperCase = false;
111 
112 			if(_arguments > 2) {
113 				xeno.common.lang.Data.ensures(xeno.common.lang.Data.isBoolean(toUpperCase), "The 'toUpperCase' is not a Boolean");
114 
115 				_toUpperCase = toUpperCase;
116 			}
117 
118 			xeno.common.lang.Data.ensures(row >= 0, "The 'row' is negative");
119 			xeno.common.lang.Data.ensures(column >= 0, "The 'column' is negative");
120 
121 			return _getAddress(row, column, _toUpperCase);
122 		},
123 
124 		/**
125 		 * @description
126 		 * <p>
127 		 * Calculates and returns the excel cell address in "A1" format.
128 		 * </p>
129 		 * <p>
130 		 * The result will be returned in an Object with "row" and "column" properties, all values are zero based.
131 		 * </p>
132 		 * 
133 		 * @example
134 		 * <b>Examples</b>
135 		 * 
136 		 * xeno.common.office.ExcelCellCalculator.getIndex("c9") = { row: 8, column: 2 }
137 		 * 
138 		 * @param address
139 		 * 			The String value (case insensitive) in excel cell address (A1) format.
140 		 * 
141 		 * @return
142 		 * 			The Object with "row" and "column" properties.
143 		 * 
144 		 * @throws
145 		 * 			The Error will be thrown when the "address" is missing.<br/>
146 		 * 			Or it will be thrown when the "address" is not a String.<br/>
147 		 * 			Or it will be thrown when the "address" is an invalid excel cell address (A1) format.
148 		 * 
149 		 * @since
150 		 * 			1.0.0
151 		 */
152 		getIndex: function(address) {
153 			xeno.common.lang.Data.ensures(arguments.length > 0, "The 'address' is required, index: 0");
154 			xeno.common.lang.Data.ensures(xeno.common.lang.Data.isString(address), "The 'address' is not a String");
155 			xeno.common.lang.Data.ensures(_address.test(address), "Invalid excel cell address: " + address);
156 
157 			return _getIndex(address);
158 		}
159 
160 	};
161 
162 })();
