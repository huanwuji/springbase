1 /**
  2  * @class
  3  * <p>
  4  * This class contains some utilities designed to provide a convenience for handling the file related operations.
  5  * </p>
  6  * 
  7  * @description
  8  * <p>
  9  * The class constructor.
 10  * </p>
 11  * 
 12  * @since
 13  * 			1.0.0
 14  */
 15 xeno.common.util.FileUtils = (function () {
 16 	var _separators = ["\\", "/"];
 17 
 18 	var _safeToExecute = function(data, nullReturnVal, func) {
 19 		return xeno.common.lang.Data.isNull(data) ? nullReturnVal : func();
 20 	};
 21 
 22 	// ------------------------------
 23 	// Private Function Definitions
 24 	// ------------------------------
 25 
 26 	var _getName = function(file) {
 27 
 28 		return _safeToExecute(file, null, function() {
 29 			var ret = file;
 30 
 31 			for(var i = 0; i < _separators.length; i += 1) {
 32 				var index = ret.lastIndexOf(_separators[i]);
 33 
 34 				if(index >= 0) {
 35 					ret = ret.substring(index + 1);
 36 				}
 37 			}
 38 
 39 			return xeno.common.util.StringUtils.isBlank(ret) ? null : ret;
 40 		});
 41 	};
 42 
 43 	var _getExtName = function(file) {
 44 
 45 		return _safeToExecute(file, null, function() {
 46 			var len = file.length;
 47 			var pos = file.lastIndexOf(".");
 48 
 49 			if(pos === -1) {
 50 				return null;
 51 
 52 			} else if(pos === len - 1) {
 53 				return "";
 54 			}
 55 
 56 			var ret = file.substring(pos + 1, len);
 57 
 58 			for(var i = 0; i < _separators.length; i += 1) {
 59 
 60 				if(ret.indexOf(_separators[i]) !== -1) {
 61 					return null;
 62 				}
 63 			}
 64 
 65 			return ret;
 66 		});
 67 	};
 68 
 69 	return {
 70 
 71 		/**
 72 		 * @description
 73 		 * <p>
 74 		 * Returns the name of the "file", or returns null if there is no name.
 75 		 * </p>
 76 		 * <p>
 77 		 * This function handles both absolute and relative file paths.
 78 		 * </p>
 79 		 * 
 80 		 * @example
 81 		 * <b>Examples</b>
 82 		 * 
 83 		 * xeno.common.util.FileUtils.getName(null) = null
 84 		 * xeno.common.util.FileUtils.getName("") = null
 85 		 * xeno.common.util.FileUtils.getName("a.gif") = "a.gif"
 86 		 * xeno.common.util.FileUtils.getName("/abc/a.exe") = "a.exe"
 87 		 * xeno.common.util.FileUtils.getName("/abc/a.") = "a."
 88 		 * xeno.common.util.FileUtils.getName("/abc/a") = "a"
 89 		 * xeno.common.util.FileUtils.getName("c:\abc.gif\a.png") = "a.png"
 90 		 * 
 91 		 * @param file
 92 		 * 			The file path to be handled.
 93 		 * 
 94 		 * @return
 95 		 * 			The String.
 96 		 * 
 97 		 * @throws
 98 		 * 			The Error will be thrown when the "file" is missing.<br/>
 99 		 * 			Or it will be thrown when the "file" is not null or not a String.
100 		 * 
101 		 * @since
102 		 * 			1.0.0
103 		 */
104 		getName: function(file) {
105 			xeno.common.lang.Data.ensures(arguments.length > 0, "The 'file' is required, index: 0");
106 			xeno.common.lang.Data.ensures(xeno.common.lang.Data.isString(file, true), "The 'file' is not null or not a String");
107 
108 			return _getName(file);
109 		},
110 
111 		/**
112 		 * @description
113 		 * <p>
114 		 * Returns the extension name of the "file", or returns null if there is no extension.
115 		 * </p>
116 		 * <p>
117 		 * This function handles both absolute and relative file paths.
118 		 * </p>
119 		 * 
120 		 * @example
121 		 * <b>Examples</b>
122 		 * 
123 		 * xeno.common.util.FileUtils.getExtName(null) = null
124 		 * xeno.common.util.FileUtils.getExtName("") = null
125 		 * xeno.common.util.FileUtils.getExtName("a.gif") = "gif"
126 		 * xeno.common.util.FileUtils.getExtName("/abc/a.exe") = "exe"
127 		 * xeno.common.util.FileUtils.getExtName("/abc/a.") = ""
128 		 * xeno.common.util.FileUtils.getExtName("/abc/a") = null
129 		 * xeno.common.util.FileUtils.getExtName("c:\abc.gif\a.png") = "png"
130 		 * 
131 		 * @param file
132 		 * 			The file path to be handled.
133 		 * 
134 		 * @return
135 		 * 			The String.
136 		 * 
137 		 * @throws
138 		 * 			The Error will be thrown when the "file" is missing.<br/>
139 		 * 			Or it will be thrown when the "file" is not null or not a String.
140 		 * 
141 		 * @since
142 		 * 			1.0.0
143 		 */
144 		getExtName: function(file) {
145 			xeno.common.lang.Data.ensures(arguments.length > 0, "The 'file' is required, index: 0");
146 			xeno.common.lang.Data.ensures(xeno.common.lang.Data.isString(file, true), "The 'file' is not null or not a String");
147 
148 			return _getExtName(file);
149 		}
150 
151 	};
152 
153 })();
