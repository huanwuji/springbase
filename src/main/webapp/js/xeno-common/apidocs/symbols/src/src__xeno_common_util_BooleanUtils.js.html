1 /**
  2  * @class
  3  * <p>
  4  * This class contains some utilities designed to provide a convenience for handling the Boolean related operations.
  5  * </p>
  6  * 
  7  * @description
  8  * <p>
  9  * The class constructor.
 10  * </p>
 11  * 
 12  * @since
 13  * 			1.0.0
 14  */
 15 xeno.common.util.BooleanUtils = (function () {
 16 
 17 	// ------------------------------
 18 	// Private Function Definitions
 19 	// ------------------------------
 20 
 21 	var _parse = function(data) {
 22 
 23 		if(xeno.common.lang.Data.isBoolean(data)) {
 24 			return data;
 25 
 26 		} else if(xeno.common.lang.Data.isNumber(data)) {
 27 			return data === 1;
 28 
 29 		} else if(xeno.common.lang.Data.isString(data)) {
 30 			var val = data.toLowerCase();
 31 
 32 			return val === "true" || val === "t" || val === "yes" || val === "y" || val === "on";
 33 		}
 34 
 35 		return false;
 36 	};
 37 
 38 	return {
 39 
 40 		/**
 41 		 * @description
 42 		 * <p>
 43 		 * Parses the "data" into a Boolean.
 44 		 * </p>
 45 		 * <p>
 46 		 * Only below values will be parsed into true.
 47 		 * Others, including undefined and null, will be parsed into false.
 48 		 * 	<ul>
 49 		 * 		<li>Number: 1</li>
 50 		 * 		<li>String: "true", "t", "yes", "y", or "on" (case insensitive)</li>
 51 		 * 		<li>Boolean: true</li>
 52 		 * 	</ul>
 53 		 * </p>
 54 		 * 
 55 		 * @param data
 56 		 * 			The data to be parsed.
 57 		 * 
 58 		 * @return
 59 		 * 			The Boolean.
 60 		 * 
 61 		 * @throws
 62 		 * 			The Error will be thrown when the "data" is missing.
 63 		 * 
 64 		 * @since
 65 		 * 			1.0.0
 66 		 */
 67 		parse: function(data) {
 68 			xeno.common.lang.Data.ensures(arguments.length > 0, "The 'data' is required");
 69 
 70 			return _parse(data);
 71 		}
 72 
 73 	};
 74 
 75 })();
