1 /**
  2  * @class
  3  * <p>
  4  * This class contains some utilities designed to provide a convenience for handling the Object related operations.
  5  * </p>
  6  * 
  7  * @description
  8  * <p>
  9  * The class constructor.
 10  * </p>
 11  *

 13  * 			1.1.0
 14  */
 15 xeno.common.util.ObjectUtils = (function () {
 16 
 17 	// ------------------------------
 18 	// Private Function Definitions
 19 	// ------------------------------
 20 
 21 	var _hasProperty = function(obj, name) {
 22 		var target = obj[name];
 23 
 24 		return !xeno.common.lang.Data.isUndefined(target) && !xeno.common.lang.Data.isFunction(target);
 25 	};
 26 
 27 	var _hasFunction = function(obj, name) {
 28 		return xeno.common.lang.Data.isFunction(obj[name]);
 29 	};
 30 
 31 	return {
 32 
 33 		/**
 34 		 * @description
 35 		 * <p>
 36 		 * Tests whether the "obj" has the named property or not.
 37 		 * </p>
 38 		 * 
 39 		 * @example
 40 		 * <b>Examples</b>
 41 		 * 
 42 		 * Assume the "obj" is: { id: 1, name: null, func: function(a, b) {} }
 43 		 * 
 44 		 * xeno.common.util.ObjectUtils.hasProperty(obj, "id") = true
 45 		 * xeno.common.util.ObjectUtils.hasProperty(obj, "name") = true
 46 		 * xeno.common.util.ObjectUtils.hasProperty(obj, "age") = false
 47 		 * xeno.common.util.ObjectUtils.hasProperty(obj, "func") = false
 48 		 * 
 49 		 * @param obj
 50 		 * 			The Object to be tested.
 51 		 * @param name
 52 		 * 			The property name.
 53 		 * 
 54 		 * @return
 55 		 * 			The Boolean.
 56 		 * 
 57 		 * @throws
 58 		 * 			The Error will be thrown when the "obj" or the "name" is missing.<br/>
 59 		 * 			Or it will be thrown when the "obj" is undefined or null.<br/>
 60 		 * 			Or it will be thrown when the "name" is not a String.<br/>
 61 		 * 			Or it will be thrown when the "name" is a blank String.
 62 		 * 
 63 		 * @since
 64 		 * 			1.1.0
 65 		 */
 66 		hasProperty: function(obj, name) {
 67 			var _arguments = arguments.length;
 68 
 69 			xeno.common.lang.Data.ensures(_arguments > 0, "The 'obj' is required, index: 0");
 70 			xeno.common.lang.Data.ensures(_arguments > 1, "The 'name' is required, index: 1");
 71 			xeno.common.lang.Data.ensures(!xeno.common.lang.Data.isUndefinedOrNull(obj), "The 'obj' is undefined or null");
 72 			xeno.common.lang.Data.ensures(xeno.common.lang.Data.isString(name), "The 'name' is not a String");
 73 			xeno.common.lang.Data.ensures(!xeno.common.util.StringUtils.isBlank(name), "The 'name' is a blank String");
 74 
 75 			return _hasProperty(obj, name);
 76 		},
 77 
 78 		/**
 79 		 * @description
 80 		 * <p>
 81 		 * Tests whether the "obj" has the named function or not.
 82 		 * </p>
 83 		 * 
 84 		 * @example
 85 		 * <b>Examples</b>
 86 		 * 
 87 		 * Assume the "obj" is: { id: 1, name: null, func: function(a, b) {} }
 88 		 * 
 89 		 * xeno.common.util.ObjectUtils.hasFunction(obj, "id") = false
 90 		 * xeno.common.util.ObjectUtils.hasFunction(obj, "name") = false
 91 		 * xeno.common.util.ObjectUtils.hasFunction(obj, "age") = false
 92 		 * xeno.common.util.ObjectUtils.hasFunction(obj, "func") = true
 93 		 * 
 94 		 * @param obj
 95 		 * 			The Object to be tested.
 96 		 * @param name
 97 		 * 			The function name.
 98 		 * 
 99 		 * @return
100 		 * 			The Boolean.
101 		 * 
102 		 * @throws
103 		 * 			The Error will be thrown when the "obj" or the "name" is missing.<br/>
104 		 * 			Or it will be thrown when the "obj" is undefined or null.<br/>
105 		 * 			Or it will be thrown when the "name" is not a String.<br/>
106 		 * 			Or it will be thrown when the "name" is a blank String.
107 		 * 
108 		 * @since
109 		 * 			1.1.0
110 		 */
111 		hasFunction: function(obj, name) {
112 			var _arguments = arguments.length;
113 
114 			xeno.common.lang.Data.ensures(_arguments > 0, "The 'obj' is required, index: 0");
115 			xeno.common.lang.Data.ensures(_arguments > 1, "The 'name' is required, index: 1");
116 			xeno.common.lang.Data.ensures(!xeno.common.lang.Data.isUndefinedOrNull(obj), "The 'obj' is undefined or null");
117 			xeno.common.lang.Data.ensures(xeno.common.lang.Data.isString(name), "The 'name' is not a String");
118 			xeno.common.lang.Data.ensures(!xeno.common.util.StringUtils.isBlank(name), "The 'name' is a blank String");
119 
120 			return _hasFunction(obj, name);
121 		}
122 
123 	};
