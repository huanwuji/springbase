1 /**
  2  * @class
  3  * <p>
  4  * This class contains some utilities designed to provide a convenience for handling the random related operations.
  5  * </p>
  6  * 
  7  * @description
  8  * <p>
  9  * The class constructor.
 10  * </p>
 11  * 
 12  * @since
 13  * 			1.0.0
 14  */
 15 xeno.common.util.RandomUtils = (function () {
 16 
 17 	// ------------------------------
 18 	// Private Function Definitions
 19 	// ------------------------------
 20 
 21 	var _generateInt = function(min, max) {
 22 		var _min = Math.floor(min);
 23 		var _max = Math.floor(max);
 24 
 25 		return Math.floor(Math.random() * (_max - _min + 1)) + _min;
 26 	};
 27 
 28 	var _generateBoolean = function() {
 29 		return _generateInt(0, 1) === 0;
 30 	};
 31 
 32 	var _generateString = function(length, candidate) {
 33 		var len = Math.floor(length);
 34 		var ret = "";
 35 
 36 		for(var i = 0; i < len; i += 1) {
 37 			var index = _generateInt(0, candidate.length - 1);
 38 
 39 			ret += candidate.charAt(index);
 40 		}
 41 
 42 		return ret;
 43 	};
 44 
 45 	var _randomizeItems = function(arr) {
 46 
 47 		if(xeno.common.lang.Data.isNull(arr)) {
 48 			return null;
 49 		}
 50 
 51 		var ret = [];
 52 		var tmp = arr.concat();
 53 
 54 		for(var i = 0; i < tmp.length; i += 1) {
 55 			var index = _generateInt(0, tmp.length - 1);
 56 			ret.push(tmp[index]);
 57 			tmp.splice(index, 1);
 58 
 59 			i -= 1;
 60 		}
 61 
 62 		return ret;
 63 	};
 64 
 65 	return {
 66 
 67 		/**
 68 		 * @description
 69 		 * <p>
 70 		 * Generates and returns an integer Number between (inclusive) the "min" and "max" in randomly.
 71 		 * </p>
 72 		 * <p>
 73 		 * The "min" or the "max" maybe a float value, the "Math.floor(n)" action will be processed first to convert it into an integer Number.
 74 		 * </p>
 75 		 * 
 76 		 * @param min
 77 		 * 			The minimum value.
 78 		 * @param max
 79 		 * 			The maximum value.
 80 		 * 
 81 		 * @return
 82 		 * 			The Number, integer.
 83 		 * 
 84 		 * @throws
 85 		 * 			The Error will be thrown when the "min" or the "max" is missing.<br/>
 86 		 * 			Or it will be thrown when the "min" or the "max" is not a Number.<br/>
 87 		 * 			Or it will be thrown when the "min" is greater than the "max".
 88 		 * 
 89 		 * @since
 90 		 * 			1.0.0
 91 		 */
 92 		generateInt: function(min, max) {
 93 			var _arguments = arguments.length;
 94 
 95 			xeno.common.lang.Data.ensures(_arguments > 0, "The 'min' is required, index: 0");
 96 			xeno.common.lang.Data.ensures(_arguments > 1, "The 'max' is required, index: 1");
 97 			xeno.common.lang.Data.ensures(xeno.common.lang.Data.isNumber(min), "The 'min' is not a Number");
 98 			xeno.common.lang.Data.ensures(xeno.common.lang.Data.isNumber(max), "The 'max' is not a Number");
 99 			xeno.common.lang.Data.ensures(min <= max, "The 'min' is greater than the 'max'");
100 
101 			return _generateInt(min, max);
102 		},
103 
104 		/**
105 		 * @description
106 		 * <p>
107 		 * Generates and returns a Boolean in randomly.
108 		 * </p>
109 		 * 
110 		 * @return
111 		 * 			The Boolean.
112 		 * 
113 		 * @since
114 		 * 			1.0.0
115 		 */
116 		generateBoolean: function() {
117 			return _generateBoolean();
118 		},
119 
120 		/**
121 		 * @description
122 		 * <p>
123 		 * Generates and returns a String in randomly, you could pass the "candidate" to generate your random String.
124 		 * </p>
125 		 * <p>
126 		 * The "length" maybe a float value, the "Math.floor(n)" action will be processed first to convert it into an integer Number.
127 		 * </p>
128 		 * 
129 		 * @param length
130 		 * 			The length of the String to generate.
131 		 * @param candidate
132 		 * 			The candidate to use.<br/>
133 		 * 			The default value is: "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ".
134 		 * 
135 		 * @return
136 		 * 			The String.
137 		 * 
138 		 * @throws
139 		 * 			The Error will be thrown when the "length" is missing.<br/>
140 		 * 			Or it will be thrown when the "length" is not a Number.<br/>
141 		 * 			Or it will be thrown when the "candidate" is not a String.<br/>
142 		 * 			Or it will be thrown when the "length" is negative.<br/>
143 		 * 			Or it will be thrown when the "candidate" is an empty String.
144 		 * 
145 		 * @since
146 		 * 			1.0.0
147 		 */
148 		generateString: function(length, candidate) {
149 			var _arguments = arguments.length;
150 
151 			xeno.common.lang.Data.ensures(_arguments > 0, "The 'length' is required, index: 0");
152 			xeno.common.lang.Data.ensures(xeno.common.lang.Data.isNumber(length), "The 'length' is not a Number");
153 
154 			var _candidate = "0123456789ABCDEFGHIJKLMNOPQRSTUVWXYZ";
155 
156 			if(_arguments > 1) {
157 				xeno.common.lang.Data.ensures(xeno.common.lang.Data.isString(candidate), "The 'candidate' is not a String");
158 
159 				_candidate = candidate;
160 			}
161 
162 			xeno.common.lang.Data.ensures(length >= 0, "The 'length' is negative");
163 			xeno.common.lang.Data.ensures(_candidate.length > 0, "The 'candidate' is an empty String");
164 
165 			return _generateString(length, _candidate);
166 		},
167 
168 		/**
169 		 * @description
170 		 * <p>
171 		 * Randomizes each item in the Array and returns a new Array instance.
172 		 * </p>
173 		 * <p>
174 		 * The original Array will not be edited, but if data in the "arr" has a deep reference, this relationship will also bring to the copied Array as well.
175 		 * </p>
176 		 * 
177 		 * @param arr
178 		 * 			The Array to be handled.
179 		 * 
180 		 * @return
181 		 * 			The Array.
182 		 * 
183 		 * @throws
184 		 * 			The Error will be thrown when the "arr" is missing.<br/>
185 		 * 			Or it will be thrown when the "arr" is not null or not an Array.
186 		 * 
187 		 * @since
188 		 * 			1.0.0
189 		 */
190 		randomizeItems: function(arr) {
191 			xeno.common.lang.Data.ensures(arguments.length > 0, "The 'arr' is required, index: 0");
192 			xeno.common.lang.Data.ensures(xeno.common.lang.Data.isArray(arr, true), "The 'arr' is not null or not an Array");
193 
194 			return _randomizeItems(arr);
195 		}
196 
197 	};
198 
199 })();
