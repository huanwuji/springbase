1 /**
  2  * @class
  3  * <p>
  4  * This class has implemented the globally unique identifier (GUID) related operations.
  5  * </p>
  6  * 
  7  * @description
  8  * <p>
  9  * The class constructor.
 10  * </p>
 11  * 
 12  * @since
 13  * 			1.0.0
 14  */
 15 xeno.common.lang.GUID = (function () {
 16 	var _count = 0;
 17 	var _guid = new RegExp("^(?:[A-F0-9]{8})(?:-[A-F0-9]{4}){3}-(?:[A-F0-9]{12})$", "i");
 18 
 19 	// ------------------------------
 20 	// Private Function Definitions
 21 	// ------------------------------
 22 
 23 	var _newValue = function(toUpperCase) {
 24 		var i1 = new Date().getTime();
 25 		var i2 = navigator.appCodeName + navigator.appName + navigator.appVersion + navigator.platform + navigator.userAgent;
 26 		var i3 = Math.random() * Number.MAX_VALUE;
 27 		var rs = xeno.common.util.CryptoUtils.md5Hex(i1 + i2 + i3 + _count++, toUpperCase);
 28 
 29 		return rs.substring(0, 8) + "-" + rs.substring(8, 12) + "-" + rs.substring(12, 16) + "-" + rs.substring(16, 20) + "-" + rs.substring(20);
 30 	};
 31 
 32 	var _emptyValue = function() {
 33 		return "00000000-0000-0000-0000-000000000000";
 34 	};
 35 
 36 	var _test = function(guid) {
 37 		return _guid.test(guid);
 38 	};
 39 
 40 	return {
 41 
 42 		/**
 43 		 * @description
 44 		 * <p>
 45 		 * Generates and returns a new GUID value.
 46 		 * </p>
 47 		 * <p>
 48 		 * The format of the value is: ########-####-####-####-############.
 49 		 * </p>
 50 		 * 
 51 		 * @param toUpperCase
 52 		 * 			A Boolean value to decide whether all characters in the return value are in upper case or not.<br/>
 53 		 * 			The default value is: false.
 54 		 * 
 55 		 * @return
 56 		 * 			The String.
 57 		 * 
 58 		 * @throws
 59 		 * 			The Error will be thrown when the "toUpperCase" is not a Boolean.
 60 		 * 
 61 		 * @since
 62 		 * 			1.0.0
 63 		 */
 64 		newValue: function(toUpperCase) {
 65 			var _toUpperCase = false;
 66 
 67 			if(arguments.length > 0) {
 68 				xeno.common.lang.Data.ensures(xeno.common.lang.Data.isBoolean(toUpperCase), "The 'toUpperCase' is not a Boolean");
 69 
 70 				_toUpperCase = toUpperCase;
 71 			}
 72 
 73 			return _newValue(_toUpperCase);
 74 		},
 75 
 76 		/**
 77 		 * @description
 78 		 * <p>
 79 		 * Returns the empty GUID value: 00000000-0000-0000-0000-000000000000.
 80 		 * </p>
 81 		 * 
 82 		 * @return
 83 		 * 			The String.
 84 		 * 
 85 		 * @since
 86 		 * 			1.0.0
 87 		 */
 88 		emptyValue: function() {
 89 			return _emptyValue();
 90 		},
 91 
 92 		/**
 93 		 * @description
 94 		 * <p>
 95 		 * Tests whether the "guid" is a valid GUID value or not.
 96 		 * </p>
 97 		 * <p>
 98 		 * The function also handles null "guid" and returns false.
 99 		 * </p>
100 		 * 
101 		 * @param guid
102 		 * 			The String to be tested.
103 		 * 
104 		 * @return
105 		 * 			The Boolean.
106 		 * 
107 		 * @throws
108 		 * 			The Error will be thrown when the "guid" is missing.<br/>
109 		 * 			Or it will be thrown when the "guid" is not null or not a String.
110 		 * 
111 		 * @since
112 		 * 			1.0.0
113 		 */
114 		test: function(guid) {
115 			xeno.common.lang.Data.ensures(arguments.length > 0, "The 'guid' is required, index: 0");
116 			xeno.common.lang.Data.ensures(xeno.common.lang.Data.isString(guid, true), "The 'guid' is not null or not a String");
117 
118 			return _test(guid);
119 		}
120 
121 	};
122 
123 })();
